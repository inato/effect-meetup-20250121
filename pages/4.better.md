function UserPreferences() {

  const favoriteValue = watch();
  const showMessage = favoriteValue.includes("Angular");

  return (
    <>
      <display.favorite label="Favorite framework" />
      {showMessage && <M.Alert>Really? Still in 2025?</M.Alert>}
    </>
  );
}
